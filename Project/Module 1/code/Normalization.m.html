close; clear; clc;
rgbSource = imread('Target2.png');
rgbTarget = imread('Source.png');
labSource = rgb2lab(rgbSource);
lSource = labSource(:,:,1);
aSource = labSource(:,:,2);
bSource = labSource(:,:,3);
%meanSource = 1/N * Sum(a[i][j]) for each i,j
%stdDevSource = sqrt(1/(N-1) * Sum((a[i][j] - Mu)^2)) for each i,j
% (in here N is the number of pixels, 20*20 in the viola jones case)
meanlSource = mean2(lSource);
meanaSource = mean2(aSource);
meanbSource = mean2(bSource);
stdlSource = std2(lSource);
stdaSource = std2(aSource);
stdbSource = std2(bSource);
% somme = 0;
% for i = 1:512
% for j = 1:512
% somme = somme + lSource(i,j);
% end
% end
% meanSource2 = somme/(512*512);
labTarget = rgb2lab(rgbTarget);
lTarget = labTarget(:,:,1);
aTarget = labTarget(:,:,2);
bTarget = labTarget(:,:,3);
meanlTarget = mean2(lTarget);
meanaTarget = mean2(aTarget);
meanbTarget = mean2(bTarget);
stdlTarget = std2(lTarget);
stdaTarget = std2(aTarget);
stdbTarget = std2(bTarget);
% figure(2);
% R = rgb(:,:,1);
% G = rgb(:,:,2);
% B = rgb(:,:,3);
% [L,a,b] = RGB2Lab(R,G,B);
% LAB = cat(3,L,a,b);
% imshow(LAB(:,:,1),[0 100]);
%Normalization
for i = 1:512
for j = 1:512
lMapped(i,j) = ((lSource(i,j)-meanlSource)/stdlSource)*stdlTarget+meanlTarget;
aMapped(i,j) = ((aSource(i,j)-meanaSource)/stdaSource)*stdaTarget+meanaTarget;
bMapped(i,j) = ((bSource(i,j)-meanbSource)/stdbSource)*stdbTarget+meanbTarget;
end
end
labMapped = cat(3,lMapped,aMapped,bMapped);
rgbMapped = lab2rgb(labMapped);
figure('name','Color normalization with Reihnard method');
subplot(1,3,1);imshow(rgbSource);title('Source');
subplot(1,3,2);imshow(rgbTarget);title('Target');
subplot(1,3,3);imshow(rgbMapped);title('Mapped');
